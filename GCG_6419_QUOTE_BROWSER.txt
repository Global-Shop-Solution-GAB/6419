SELECT QUOTE_NO AS "Quote Number", CUSTOMER AS "Customer", IF(DATE_DUE > '000000',  IF(SUBSTRING(DATE_DUE,5,2)<80,CONVERT('20'+SUBSTRING(DATE_DUE,5,2)+'-'+SUBSTRING(DATE_DUE,1,2)+'-'+SUBSTRING(DATE_DUE,3,2),SQL_DATE),CONVERT('19'+SUBSTRING(DATE_DUE,5,2)+'-'+SUBSTRING(DATE_DUE,1,2)+'-'+SUBSTRING(DATE_DUE,3,2),SQL_DATE)), NULL) AS "Date Due", CUSTOMER_PO AS "Customer PO" ,QUOTE_HEADER.QUOTE_NO,QUOTE_HEADER.CUSTOMER,QUOTE_HEADER.DATE_DUE,QUOTE_HEADER.CUSTOMER_PO,USER_1 AS "User 1",USER_2 AS "User 2", USER_3 AS  "User 3", User_4 AS "User 4",User_5 AS "User 5" FROM QUOTE_HEADER WHERE QUOTE_NO <> '' AND RECORD_NO = '  A' ORDER BY "Quote Number"
